// 1. Define the base field
k := Rationals();

// 2. Define the polynomial ring
// We include an auxiliary variable (t), primal variables (x), and dual variables (y).
// The "elim", 7 block ordering optimizes the Groebner basis computation 
// for eliminating the first 7 variables (t and x0..x5).
R<t, x0,x1,x2,x3,x4,x5, y0,y1,y2,y3,y4,y5> := PolynomialRing(k, 13, "elim", 7);

// 3. Define the equation of the hypersurface F
F := (x0*x1 - x2*x3)*(x4 + x5) + x0*x4^2 + x1*x4*x5 + x2*x5^2;

// 4. Compute the gradient vector (partial derivatives of F)
gradF := [ Derivative(F, x0), 
           Derivative(F, x1), 
           Derivative(F, x2),
           Derivative(F, x3), 
           Derivative(F, x4), 
           Derivative(F, x5) ];

// 5. Create the relations for the gradient map
// The coordinates of the tangent hyperplane are proportional to the gradient.
// We use the auxiliary variable t to handle the proportionality: y_i = t * (dF/dx_i)
rels := [ y0 - t*gradF[1],
          y1 - t*gradF[2],
          y2 - t*gradF[3],
          y3 - t*gradF[4],
          y4 - t*gradF[5],
          y5 - t*gradF[6] ];

// 6. Form the ideal generated by F and the gradient relations
I := ideal< R | F, rels >;

// 7. Compute the dual variety
// We eliminate t and x0..x5 by keeping only the dual variables y0..y5.
DualIdeal := EliminationIdeal(I, {y0, y1, y2, y3, y4, y5});

// 8. Output the result
print "The ideal defining the dual variety X^* is:";
print DualIdeal;
